## Install

```bash
npm install --save sharkio
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'sharkio'
import 'sharkio/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
